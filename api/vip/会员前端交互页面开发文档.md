# 会员前端交互页面开发文档

## 1. 文档概述

本文档详细描述了会员系统前端交互的完整实现方案，包括会员状态显示、升级交互、权限控制等功能的前端开发规范。

### 1.1 功能范围
- 会员状态动态显示
- 会员升级交互流程
- 会员权益展示
- 下载权限控制
- 会员码兑换界面
- 会员信息管理

### 1.2 技术栈
- HTML5 + CSS3 + JavaScript (ES6+)
- Tailwind CSS 框架
- Font Awesome 图标库
- Fetch API 进行数据交互

## 4. JavaScript 实现逻辑

### 4.1 会员状态管理类
```javascript
class MembershipManager {
    constructor() {
        this.currentMembership = null;
        this.init();
    }
    
    async init() {
        await this.loadMembershipStatus();
        this.bindEvents();
    }
    
    // 加载会员状态
    async loadMembershipStatus() {
        try {
            const response = await fetch('/api/vip/get_user_membership.php', {
                credentials: 'include'
            });
            const result = await response.json();
            
            if (result.success) {
                this.currentMembership = result.data.membership;
                this.updateMembershipDisplay();
            } else {
                console.error('获取会员状态失败:', result.message);
                this.showDefaultMembership();
            }
        } catch (error) {
            console.error('网络错误:', error);
            this.showDefaultMembership();
        }
    }
    
    // 更新会员状态显示
    updateMembershipDisplay() {
        const membership = this.currentMembership;
        const membershipType = membership.type || 'free';
        const config = MEMBERSHIP_TYPES[membershipType];
        
        // 更新侧边栏状态
        this.updateSidebarStatus(config, membership);
        
        // 更新权益卡片
        this.updateBenefitsCard(membershipType, membership);
        
        // 更新升级按钮
        this.updateUpgradeButtons(membershipType);
    }
    
    // 更新侧边栏状态
    updateSidebarStatus(config, membership) {
        const statusEl = document.getElementById('membership-status');
        const badgeEl = document.getElementById('membership-badge');
        const expiryEl = document.getElementById('membership-expiry');
        
        if (statusEl) {
            statusEl.textContent = config.name;
            statusEl.className = `text-sm ${config.color}`;
        }
        
        if (badgeEl) {
            badgeEl.textContent = config.badge;
            badgeEl.className = `px-2 py-1 text-xs rounded-full ${config.bgColor} ${config.color}`;
            badgeEl.classList.remove('hidden');
        }
        
        // 显示到期时间（仅月度会员）
        if (expiryEl && membership.type === 'monthly' && membership.expires_at) {
            const expiryDate = new Date(membership.expires_at);
            const daysRemaining = Math.ceil((expiryDate - new Date()) / (1000 * 60 * 60 * 24));
            
            if (daysRemaining > 0) {
                expiryEl.textContent = `剩余 ${daysRemaining} 天`;
                expiryEl.classList.remove('hidden');
            }
        }
    }
    
    // 更新权益卡片
    updateBenefitsCard(membershipType, membership) {
        // 隐藏所有权益卡片
        ['free-user-benefits', 'monthly-member-benefits', 'permanent-member-benefits'].forEach(id => {
            const el = document.getElementById(id);
            if (el) el.classList.add('hidden');
        });
        
        // 显示对应的权益卡片
        const benefitsEl = document.getElementById(`${membershipType === 'free' ? 'free-user' : membershipType + '-member'}-benefits`);
        if (benefitsEl) {
            benefitsEl.classList.remove('hidden');
        }
        
        // 更新图标
        const iconEl = document.getElementById('membership-icon');
        if (iconEl) {
            const config = MEMBERSHIP_TYPES[membershipType];
            iconEl.className = `fa ${config.icon} ${config.color}`;
        }
        
        // 更新月度会员配额信息
        if (membershipType === 'monthly' && membership.download_quota !== undefined) {
            const quotaEl = document.getElementById('remaining-downloads');
            if (quotaEl) {
                quotaEl.textContent = membership.download_quota >= 0 ? membership.download_quota : '无限制';
            }
        }
    }
    
    // 更新升级按钮
    updateUpgradeButtons(membershipType) {
        // 隐藏所有升级选项
        ['free-upgrade-options', 'monthly-upgrade-option', 'permanent-member-info'].forEach(id => {
            const el = document.getElementById(id);
            if (el) el.classList.add('hidden');
        });
        
        // 显示对应的升级选项
        let targetId;
        switch (membershipType) {
            case 'free':
                targetId = 'free-upgrade-options';
                break;
            case 'monthly':
                targetId = 'monthly-upgrade-option';
                break;
            case 'permanent':
                targetId = 'permanent-member-info';
                break;
        }
        
        const targetEl = document.getElementById(targetId);
        if (targetEl) {
            targetEl.classList.remove('hidden');
        }
    }
    
    // 绑定事件
    bindEvents() {
        // 升级按钮点击事件
        document.querySelectorAll('.upgrade-btn').forEach(btn => {
            btn.addEventListener('click', () => {
                this.showUpgradeModal();
            });
        });
        
        // 兑换按钮事件
        const redeemBtn = document.getElementById('redeem-membership-btn');
        if (redeemBtn) {
            redeemBtn.addEventListener('click', () => {
                this.redeemMembershipCode();
            });
        }
        
        // 会员码输入验证
        const codeInput = document.getElementById('membership-code-input');
        if (codeInput) {
            codeInput.addEventListener('input', (e) => {
                this.validateMembershipCode(e.target.value);
            });
        }
        
        // 模态框关闭事件
        document.getElementById('close-membership-modal')?.addEventListener('click', () => {
            this.hideUpgradeModal();
        });
        
        document.getElementById('close-success-modal')?.addEventListener('click', () => {
            this.hideSuccessModal();
        });
    }
    
    // 显示升级模态框
    showUpgradeModal() {
        const modal = document.getElementById('membership-upgrade-modal');
        if (modal) {
            modal.classList.remove('hidden');
            // 清空输入框
            const input = document.getElementById('membership-code-input');
            if (input) input.value = '';
        }
    }
    
    // 隐藏升级模态框
    hideUpgradeModal() {
        const modal = document.getElementById('membership-upgrade-modal');
        if (modal) {
            modal.classList.add('hidden');
        }
    }
    
    // 验证会员码格式
    validateMembershipCode(code) {
        const messageEl = document.getElementById('code-validation-message');
        const redeemBtn = document.getElementById('redeem-membership-btn');
        
        if (!messageEl || !redeemBtn) return;
        
        if (code.length === 0) {
            messageEl.classList.add('hidden');
            redeemBtn.disabled = true;
            return;
        }
        
        const isValid = /^[A-Z0-9]{12}$/.test(code);
        
        if (isValid) {
            messageEl.textContent = '✓ 会员码格式正确';
            messageEl.className = 'mt-2 text-sm text-green-600';
            messageEl.classList.remove('hidden');
            redeemBtn.disabled = false;
        } else {
            messageEl.textContent = '请输入12位大写字母和数字组成的会员码';
            messageEl.className = 'mt-2 text-sm text-red-600';
            messageEl.classList.remove('hidden');
            redeemBtn.disabled = true;
        }
    }
    
    // 兑换会员码
    async redeemMembershipCode() {
        const codeInput = document.getElementById('membership-code-input');
        const redeemBtn = document.getElementById('redeem-membership-btn');
        
        if (!codeInput || !redeemBtn) return;
        
        const code = codeInput.value.trim();
        if (!code) {
            this.showToast('请输入会员码', 'error');
            return;
        }
        
        // 禁用按钮，显示加载状态
        redeemBtn.disabled = true;
        redeemBtn.innerHTML = '<i class="fa fa-spinner fa-spin mr-2"></i>兑换中...';
        
        try {
            const response = await fetch('/api/vip/redeem_membership_code.php', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                credentials: 'include',
                body: JSON.stringify({ code: code })
            });
            
            const result = await response.json();
            
            if (result.success) {
                this.hideUpgradeModal();
                this.showUpgradeSuccess(result.data);
                // 重新加载会员状态
                await this.loadMembershipStatus();
            } else {
                this.showToast(result.message || '兑换失败', 'error');
            }
        } catch (error) {
            console.error('兑换失败:', error);
            this.showToast('网络错误，请稍后重试', 'error');
        } finally {
            // 恢复按钮状态
            redeemBtn.disabled = false;
            redeemBtn.innerHTML = '<i class="fa fa-check mr-2"></i>立即兑换';
        }
    }
    
    // 显示升级成功
    showUpgradeSuccess(data) {
        const modal = document.getElementById('upgrade-success-modal');
        const messageEl = document.getElementById('upgrade-success-message');
        const summaryEl = document.getElementById('upgrade-benefits-summary');
        
        if (!modal) return;
        
        // 更新成功消息
        if (messageEl) {
            const membershipName = MEMBERSHIP_TYPES[data.membership_type]?.name || '会员';
            messageEl.textContent = `恭喜您成功升级为${membershipName}！`;
        }
        
        // 更新权益摘要
        if (summaryEl) {
            let benefitsHtml = '';
            if (data.membership_type === 'monthly') {
                benefitsHtml = `
                    <h4 class="font-medium mb-2">您的权益</h4>
                    <ul class="text-sm text-gray-600 space-y-1">
                        <li>• 每月10次超清下载</li>
                        <li>• 无水印壁纸</li>
                        <li>• 优先客服支持</li>
                    </ul>
                `;
            } else if (data.membership_type === 'permanent') {
                benefitsHtml = `
                    <h4 class="font-medium mb-2">您的权益</h4>
                    <ul class="text-sm text-gray-600 space-y-1">
                        <li>• 无限制下载</li>
                        <li>• 全站壁纸无水印</li>
                        <li>• 专属会员标识</li>
                        <li>• VIP客服通道</li>
                    </ul>
                `;
            }
            summaryEl.innerHTML = benefitsHtml;
        }
        
        modal.classList.remove('hidden');
    }
    
    // 隐藏成功模态框
    hideSuccessModal() {
        const modal = document.getElementById('upgrade-success-modal');
        if (modal) {
            modal.classList.add('hidden');
        }
    }
    
    // 显示默认会员状态
    showDefaultMembership() {
        this.currentMembership = { type: 'free' };
        this.updateMembershipDisplay();
    }
    
    // 显示提示消息
    showToast(message, type = 'info') {
        // 这里可以集成现有的 toast 组件
        console.log(`${type.toUpperCase()}: ${message}`);
        alert(message); // 临时使用 alert，建议替换为更好的 toast 组件
    }
}
```

## 8. 性能优化

### 8.1 数据缓存
```javascript
class MembershipCache {
    constructor() {
        this.cache = new Map();
        this.cacheTimeout = 5 * 60 * 1000; // 5分钟缓存
    }
    
    set(key, data) {
        this.cache.set(key, {
            data,
            timestamp: Date.now()
        });
    }
    
    get(key) {
        const item = this.cache.get(key);
        if (!item) return null;
        
        if (Date.now() - item.timestamp > this.cacheTimeout) {
            this.cache.delete(key);
            return null;
        }
        
        return item.data;
    }
}
```
